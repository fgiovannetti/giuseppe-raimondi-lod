# CQ6 The texts contained in the notebook;

SELECT DISTINCT ?text
	WHERE {
      ?notebook ecrm:P2_has_type ficlitdlo:notebook ;
                ecrm:P1_is_identified_by [ecrm:P2_has_type ficlitdlo:inventory-number ; rdfs:label "RDq 303"] .
      ?notebook ecrm:P128_carries ?text .
    }

# CQ7 The dates on which the texts were created;

SELECT DISTINCT ?text ?date_of_creation
	WHERE {
      ?text efrbroo:R17i_was_created_by ?creation .
      ?creation ecrm:P4_has_time-span ?time-span .
      ?time-span rdfs:label ?date_of_creation .
    }
